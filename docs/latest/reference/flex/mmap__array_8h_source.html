164  void resize(size_t size) {

166  return;
167  }
168 
169  if (memory_only_) {
170  if (size == 0) {
171  reset();
172  } else {
173  T* new_data = static_cast<T*>(
174  mmap(NULL, size * sizeof(T), PROT_READ | PROT_WRITE,
175  MAP_PRIVATE | MAP_ANONYMOUS | MAP_NORESERVE, -1, 0));
176  if (new_data == MAP_FAILED) {
177  LOG(FATAL) << "mmap failed " << strerror(errno) << "..\n";
178  }
179  if (data_ != NULL) {
180  size_t copy_size = std::min(size, size_);
181  if (copy_size > 0) {
182  memcpy((void*) new_data, (void*) data_, copy_size * sizeof(T));
183  }
184  munmap(data_, size_ * sizeof(T));
185  }
186  data_ = new_data;
187  size_ = size;
188  if (fd_ != -1) {
189  filename_.clear();

192  }
193  }
194  } else {
